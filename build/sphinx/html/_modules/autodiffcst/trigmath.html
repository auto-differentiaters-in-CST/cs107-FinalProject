[docs]def sin(ad):
    if isinstance(ad, AD):
        new_val = sin(ad.val)
        der = cos(ad.val)
        return chain_rule(ad, new_val, der)
    elif isinstance(ad, int) or isinstance(ad, float):
        return math.sin(ad)
    else:
        raise TypeError("Input should be either an AD object or a number.")
